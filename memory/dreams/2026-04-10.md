```bash

./scripts/context-analyze.sh
```

```bash

./scripts/context-analyze.sh

# 仅生成 WLB context
./scripts/context-analyze.sh --agent wlb

# 仅生成 GSD context
./scripts/context-analyze.sh --agent gsd
```
